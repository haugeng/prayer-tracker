<script>

    let timerInterval, elapsedSeconds = 0, isPaused = false, startTime, wakeLock = null;

    // LocalStorage management for main pointers

    const mainPointersEl = document.getElementById('main-pointers');

    mainPointersEl.value = localStorage.getItem('mainPointers') || "";

    mainPointersEl.addEventListener('input', () => localStorage.setItem('mainPointers', mainPointersEl.value));

    function showPage(page) {

        ['setup-page', 'manual-page', 'history-page'].forEach(p => document.getElementById(p).classList.add('hidden'));

        document.getElementById(page + '-page').classList.remove('hidden');

        if(page === 'history') loadHistory();

        if(page === 'manual') prepManualLog();

    }

    // Wake Lock to prevent mobile screen sleep

    async function requestWakeLock() {

        if ('wakeLock' in navigator) {

            try {

                wakeLock = await navigator.wakeLock.request('screen');

                document.getElementById('wake-status').innerText = "☀️ Screen stay-awake active";

            } catch (err) { console.error(`Wake Lock Error: ${err.message}`); }

        }

    }

    function releaseWakeLock() {

        if (wakeLock) {

            wakeLock.release().then(() => {

                wakeLock = null;

                document.getElementById('wake-status').innerText = "Wake-lock inactive";

            });

        }

    }

    // Live Timer Engine

    function startSession() {

        if (!isPaused) {

            startTime = new Date();

            requestWakeLock();

        }

        isPaused = false;

        document.getElementById('start-btn').classList.add('hidden');

        document.getElementById('pause-btn').classList.remove('hidden');

        document.getElementById('stop-btn').classList.remove('hidden');

        document.getElementById('session-status').innerText = "Session in progress...";

        timerInterval = setInterval(() => { elapsedSeconds++; updateTimerDisplay(); }, 1000);

    }

    function togglePause() {

        if (!isPaused) {

            clearInterval(timerInterval);

            isPaused = true;

            document.getElementById('pause-btn').innerText = "Resume";

            document.getElementById('session-status').innerText = "Session Paused";

            releaseWakeLock();

        } else {

            document.getElementById('pause-btn').innerText = "Pause";

            startSession();

        }

    }

    function stopSession() {

        clearInterval(timerInterval);

        const endTime = new Date();

        saveRecord(startTime.toISOString(), endTime.toISOString(), elapsedSeconds, mainPointersEl.value);

        releaseWakeLock();

       

        elapsedSeconds = 0; isPaused = false;

        updateTimerDisplay();

        document.getElementById('start-btn').classList.remove('hidden');

        document.getElementById('pause-btn').classList.add('hidden');

        document.getElementById('stop-btn').classList.add('hidden');

        document.getElementById('session-status').innerText = "Session Saved!";

    }

    function updateTimerDisplay() {

        const hrs = Math.floor(elapsedSeconds / 3600).toString().padStart(2, '0');

        const mins = Math.floor((elapsedSeconds % 3600) / 60).toString().padStart(2, '0');

        const secs = (elapsedSeconds % 60).toString().padStart(2, '0');

        document.getElementById('display-time').innerText = `${hrs}:${mins}:${secs}`;

    }

    // Manual Entry Logic

    function prepManualLog() {

        const now = new Date();

        const fmt = d => new Date(d.getTime() - d.getTimezoneOffset()*60000).toISOString().slice(0,16);

        document.getElementById('manual-start').value = fmt(new Date(now - 1800000));

        document.getElementById('manual-end').value = fmt(now);

        document.getElementById('manual-pointers').value = mainPointersEl.value;

    }

    function saveManualLog() {

        const s = new Date(document.getElementById('manual-start').value);

        const e = new Date(document.getElementById('manual-end').value);

        const diff = Math.floor((e - s) / 1000);

        if (diff <= 0) return alert("Error: End time must be after start time.");

        saveRecord(s.toISOString(), e.toISOString(), diff, document.getElementById('manual-pointers').value);

        alert("Manual record saved.");

        showPage('history');

    }

    // History and Data Persistence

    function saveRecord(start, end, duration, pointers) {

        let history = JSON.parse(localStorage.getItem('prayerHistory') || "[]");

        history.unshift({ id: Date.now(), start, end, duration, pointers });

        localStorage.setItem('prayerHistory', JSON.stringify(history));

    }

    function loadHistory() {

        const history = JSON.parse(localStorage.getItem('prayerHistory') || "[]");

        const list = document.getElementById('history-list');

        list.innerHTML = "";

       

        let totalMins = 0;

        history.forEach(item => {

            totalMins += (item.duration / 60);

            const startStr = new Date(item.start).toLocaleString([], {month:'short', day:'numeric', hour:'2-digit', minute:'2-digit'});

            const row = document.createElement('tr');

            row.innerHTML = `

                <td>

                    <div style="display:flex; justify-content:space-between; align-items:center;">

                        <strong>${startStr}</strong>

                        <span>${Math.round(item.duration / 60)}m <button class="btn" style="color:var(--error); padding:0 5px;" onclick="deleteRecord(${item.id})">✕</button></span>

                    </div>

                    <div class="pointer-cell" contenteditable="true" onblur="updateRecord(${item.id}, this.innerText)">${item.pointers}</div>

                </td>

            `;

            list.appendChild(row);

        });

        document.getElementById('stat-total-mins').innerText = Math.floor(totalMins);

        document.getElementById('stat-count').innerText = history.length;

    }

    function updateRecord(id, val) {

        let history = JSON.parse(localStorage.getItem('prayerHistory') || "[]");

        const i = history.findIndex(h => h.id === id);

        if(i !== -1) { history[i].pointers = val; localStorage.setItem('prayerHistory', JSON.stringify(history)); }

    }

    function deleteRecord(id) {

        if(confirm("Permanently delete this record?")) {

            let history = JSON.parse(localStorage.getItem('prayerHistory') || "[]");

            localStorage.setItem('prayerHistory', JSON.stringify(history.filter(h => h.id !== id)));

            loadHistory();

        }

    }

    // Export & Backup

    function exportToCSV() {

        const history = JSON.parse(localStorage.getItem('prayerHistory') || "[]");

        let csv = "Start Time,End Time,Minutes,Pointers\n" + history.map(i =>

            `${new Date(i.start).toLocaleString().replace(/,/g,'')},${new Date(i.end).toLocaleString().replace(/,/g,'')},${Math.round(i.duration/60)},"${i.pointers.replace(/"/g,'""')}"`

        ).join("\n");

        downloadFile(csv, 'prayer_log_export.csv', 'text/csv');

    }

    function backupJSON() {

        const data = localStorage.getItem('prayerHistory') || "[]";

        downloadFile(data, 'prayer_tracker_backup.json', 'application/json');

    }

    function restoreJSON(event) {

        const file = event.target.files[0];

        if(!file) return;

        const reader = new FileReader();

        reader.onload = e => {

            if(confirm("Merge this with your current history?")) {

                let current = JSON.parse(localStorage.getItem('prayerHistory') || "[]");

                let imported = JSON.parse(e.target.result);

                let merged = [...imported, ...current].filter((v,i,a)=>a.findIndex(t=>(t.id === v.id))===i);

                localStorage.setItem('prayerHistory', JSON.stringify(merged));

                loadHistory();

            }

        };

        reader.readAsText(file);

    }

    function downloadFile(content, name, type) {

        const blob = new Blob([content], {type});

        const url = URL.createObjectURL(blob);

        const a = document.createElement('a');

        a.href = url; a.download = name; a.click();

        URL.revokeObjectURL(url);

    }
